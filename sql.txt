SELECT Заказчик.Код, Заказчик.Название as "Имя заказчика", Город.Название as "Город"
FROM Заказчик JOIN Город ON Код_города = Город.Код;

SELECT Заказчик.Код, Заказчик.Название as "Имя заказчика", Город.Название as "Город"
FROM Заказчик LEFT JOIN Город ON Код_города = Город.Код;


SELECT Заказчик.Код, Заказчик.Название as "Имя заказчика", Город.Название as "Город"
FROM Заказчик RIGHT JOIN Город ON Код_города = Город.Код;
